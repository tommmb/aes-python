CREATE TABLE Users (
	id INT NOT NULL AUTO_INCREMENT,
    EMAIL VARCHAR(40) NOT NULL,
    password VARCHAR(60) NOT NULL,
    salt VARCHAR(60) NOT NULL,
    first_name VARCHAR(20) NOT NULL,
    last_name VARCHAR(20) NOT NULL,
    PRIMARY KEY(id)
);
